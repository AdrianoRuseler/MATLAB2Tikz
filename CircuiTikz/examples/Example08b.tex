\documentclass[border=3mm]{standalone}

\usepackage{tikz}
\usetikzlibrary{arrows,shapes.gates.logic.US,shapes.gates.logic.IEC,calc}
\begin{document}
	\thispagestyle{empty}
	\tikzstyle{branch}=[fill,shape=circle,minimum size=3pt,inner sep=0pt]
	\begin{tikzpicture}[label distance=2mm]
		
		\node (A) at (0,0) {$A$};
		\node (B) at (1,0) {$B$};
		\node (C) at (2,0) {$C$};
		
		\node[not gate US, draw, rotate=-90] at ($(A)+(0.5,-1)$) (NotA) {};
		\node[not gate US, draw, rotate=-90] at ($(B)+(0.5,-1)$) (NotB) {};
		\node[not gate US, draw, rotate=-90] at ($(C)+(0.5,-1)$) (NotC) {};

	\foreach \i in {A,B,C}
{
	\path (\i) -- coordinate (punt\i) (\i |- Not\i.input);
	\draw (punt\i) node[branch] {} -| (Not\i.input);
}

	
	\node[xor gate US, draw, logic gate inputs=nn] at ($(C)+(2,-2)$) (gate1) {1};
\node[xor gate US, draw, logic gate inputs=nn] at ($(gate1)+(0,-1)$) (gate2) {2};
\node[xor gate US, draw, logic gate inputs=nn] at ($(gate2)+(0,-1)$) (gate3) {3};

	\node[and gate US, draw, logic gate inputs=nnn, anchor=input 1] at ($(gate1.output)+(1,-0.5)$) (gate4) {4};	


\draw (A) |- (gate1.input 1);
\draw (NotA) |- (gate3.input 2);

\draw (B) |- (gate2.input 1);
\draw (NotB) |- (gate2.input 2);

\draw (C) |- (gate3.input 1);
\draw (NotC) |- (gate1.input 2);


		\draw (gate1.output) -- ([xshift=0.4cm]gate1.output) |- (gate4.input 1);
	\draw (gate2.output) -- ([xshift=0.5cm]gate2.output) |- (gate4.input 2);
	\draw (gate3.output) -- ([xshift=0.6cm]gate3.output) |- (gate4.input 3);
	
	\draw (gate4.output) -- ([xshift=0.5cm]gate4.output) node[above] {$Y$};	
	
	\end{tikzpicture}
